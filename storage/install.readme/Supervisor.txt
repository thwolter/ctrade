Install Supervisor
sudo apt-get install supervisor

Copy laravel.conf into to /etc/supervisor/conf.d/laravel.conf
sudo cp storage/install.readme/laravel.conf /etc/supervisor/conf.d/laravel.conf
 
Start Supervisor

sudo service supervisor start

sudo supervisorctl reread
sudo supervisorctl update
sudo supervisorctl start redis-server
sudo supervisorctl start horizon
sudo supervisorctl start echo-server
